// 取得屬性設定 var props = PropertiesService.getScriptProperties(); var PRODUCT_DB_ID = props.getProperty('PRODUCT_DB_ID') || ""; function doGet(e) { try { // 1. 開啟試算表與指定分頁 var ss = SpreadsheetApp.openById(PRODUCT_DB_ID); var sheet = ss.getSheetByName("Products"); // 請確保分頁名稱正確 // 2. 取得所有資料 (假設第一列是標題) var data = sheet.getDataRange().getValues(); var headers = data.shift(); // 移除並取得標題列 // 3. 將資料轉換為 JSON 物件陣列 var jsonOutput = data.map(function(row) { var obj = {}; headers.forEach(function(header, index) { // 對應欄位：Category, Name, PV, Link, Tag obj[header.toLowerCase()] = row[index]; }); return obj; }); // 4. 回傳 JSON 格式，並允許跨網域存取 (CORS) return ContentService.createTextOutput(JSON.stringify(jsonOutput)) .setMimeType(ContentService.MimeType.JSON); } catch (f) { return ContentService.createTextOutput(JSON.stringify({ "error": f.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
